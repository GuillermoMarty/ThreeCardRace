02 /**

08 public abstract class GamePlayer {

10     private String playerName = "Player";
11 
12     /** Indicator of interactivity. */

14 
15     /**

17     * Defaults to interactive player.
18     * @param name the name of this player

20     public GamePlayer(final String name) {
21         this(name, true);
22     }
23 
24     /**
25      * Construct a player with a specified name
26      * and interactivity.

30     public GamePlayer(final String name, final boolean interactive) {
31         this.playerName = name;
32         this.isInteractive = interactive;
33     }
34 
35     /**
36      * Access name of this player.
37      * @return the name of this player
38      */
39     public String name() {
40         return this.playerName;
41     }
42 
43     /**
44      * Access interactivity of this player.
45      * @return true if interactive; false otherwise
46      */
47     public boolean isInteractive() {
48         return this.isInteractive;
49     }
50 
51     @Override public String toString() {
52         return this.playerName;
53     }
54 }
