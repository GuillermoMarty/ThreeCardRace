069         defaultHand.sort();
070         Iterator<Card> handIterator = defaultHand.iterator();
071         assertNotNull(handIterator);
072         for (int i = 0; i < defaultHand.getSize(); i++) {
073             assertTrue(handIterator.hasNext());
074             Card c = handIterator.next();
075             assertEquals(Suit.values()[0], c.suit());
076             assertEquals(Rank.values()[i], c.rank());
077         }

125         for (int i = 0; i < Hand.HAND_SIZE; i++) {
126             assertTrue(itr.hasNext());

128         }

130         for (Card c : handy) {
131             assertTrue(cards.contains(c));
132         }

176     /** Number of sorting tests to run. */
177     private static final int NUM_SORT_TESTS = 20;
178     /**
179      * Sorting tests.
180      */
181     @Test
182     public void sortTest() {
183         Hand handy = new Hand(Hand.HAND_SIZE, new Deck(), FIVE_DIAMONDS_LOC);
184         assertEquals(handy.getCard(0), new Card(FIVE, DIAMONDS));
185         // Reverse the order of the hand.
186         for (int i = 0; i < Hand.HAND_SIZE / 2; i++) {
187             Card temp = handy.getCard(i);
188             handy.setCard(i, handy.getCard(Hand.HAND_SIZE - (1 + i)));
189             handy.setCard(Hand.HAND_SIZE - (1 + i), temp);
190         }
191         assertEquals(handy.getCard(0), new Card(NINE, DIAMONDS));
192         handy.sort();
193         assertEquals(handy.getCard(0), new Card(FIVE, DIAMONDS));
194         for (int i = 0; i < Hand.HAND_SIZE - 1; i++) {
195             assertTrue(handy.getCard(i).rank().value()
196                 < handy.getCard(i + 1).rank().value());
197         }
198         // Generate and test random hands.
199         for (int h = 0; h < NUM_SORT_TESTS; h++) {
200             Deck deck = new Deck();
201             int ci = (int) Math.random() * (Deck.DECK_SIZE - Hand.HAND_SIZE);
202             for (int i = 0; i < Hand.HAND_SIZE; i++) {
203                 handy.setCard(i, deck.getCard(ci + i));
204             }
205             handy.sort();
206             for (int i = 0; i < Hand.HAND_SIZE - 1; i++) {
207                 assertTrue(handy.getCard(i).rank().value()
208                     < handy.getCard(i + 1).rank().value());
209             }
210         }
211     }
