012 public class CardTest {
013     /** The Clubs suit. */
014     private static final Suit CLUBS = Suit.valueOf("CLUBS");
015     /** The Diamonds suit. */
016     private static final Suit DIAMONDS = Suit.valueOf("DIAMONDS");
017     /** The Hearts suit. */
018     private static final Suit HEARTS = Suit.valueOf("HEARTS");
019     /** The Spades suit. */
020     private static final Suit SPADES = Suit.valueOf("SPADES");
021     /** The Seven rank. */
022     private static final Rank SEVEN = Rank.valueOf("SEVEN");
023     /** The Jack rank. */
024     private static final Rank JACK = Rank.valueOf("JACK");
025     /** The Queen rank. */
026     private static final Rank QUEEN = Rank.valueOf("QUEEN");
027 
028     /** Queen of Hearts cardID. */
029     private static final int QH_ID
030                 = (HEARTS.ordinal() * Rank.values().length) + QUEEN.ordinal();
031 
032     /** Seven of Clubs cardID. */
033     private static final int SC_ID
034                 = (CLUBS.ordinal() * Rank.values().length) + SEVEN.ordinal();
035 
036     /**
037      * Verify default card construction and access.
038      * Exercises equals and hashCode methods.
039      */
040     @Test
041     public final void defaultCardTest() {
042         Card cardDefault = new Card();
043         assertNotNull(cardDefault.rank());
044         assertNotNull(cardDefault.suit());
045         assertEquals(cardDefault, (new Card()));
046         assertEquals(cardDefault.hashCode(), (new Card()).hashCode());
047     }
048 
049     /**
050      * Verify Queen of Hearts construction and access.
051      * Exercises equals and hashCode methods.
052      */
053     @Test
054     public final void queenOfHeartsTest() {
055         Card cardQH = new Card(QUEEN, HEARTS);
056         assertEquals(QUEEN, cardQH.rank());
057         assertEquals(HEARTS, cardQH.suit());
058         assertEquals(QH_ID, cardQH.cardID());
059         assertEquals("Queen of Hearts", cardQH.toString());
060         assertEquals(cardQH, (new Card(QH_ID)));
061         assertEquals(cardQH.hashCode(), (new Card(QH_ID)).hashCode());
062     }

068     @Test
