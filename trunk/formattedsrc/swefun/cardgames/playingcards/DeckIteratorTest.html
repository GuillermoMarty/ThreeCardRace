22     @Test
23     public void compositeTest() {
24         int numCardsPerDeck = TEST_NUM_CARDS_PER_DECK;
25         DeckIterator di = new DeckIterator(numCardsPerDeck);
26         List<Card> cardlist = new ArrayList<Card>();
27         for (int i = 0; i < numCardsPerDeck; i++) {
28             cardlist.add(new Card(i));
29         }
30         Deck initialDeck = new Deck(cardlist);
31         assertNotNull(initialDeck);
32         Deck currentDeck = null;
33         long numPermutations = 1;
34         for (int i = 2; i <= numCardsPerDeck; i++) {
35             numPermutations *= i;
36         }
37         for (int i = 0; i < numPermutations; i++) {
38             assertTrue(
39                 "hasNext() expected to be true, permutations not exhausted",
40                 di.hasNext());
41             currentDeck = di.next();
42             assertEquals(numCardsPerDeck, currentDeck.size());
43             for (int k = 0; k < numCardsPerDeck; k++) {
44                 assertTrue("Expected card not found in deck.",
45                            cardlist.contains(currentDeck.deal()));
46             }
47         }
48         assertFalse("hasNext() expected to be false, permutations exhausted",
49                     di.hasNext());
50     }
