135     /** Minimum number of displaced cards for reasonable shuffle. */
136     private static final int SHUFFLE_OK = 10;
137     /**
138      * Check "reasonableness" of shuffle.

140     @Test

142         Deck shuffled = new Deck();

154     /**
155      * Test deck mutator setCard.
156      */
157     @Test
158     public void mutateDeck() {
159         Deck defaultDeck = new Deck();
160         assertEquals(Deck.DECK_SIZE, defaultDeck.size());
161         assertEquals(new Card(ACE, CLUBS),
162                      defaultDeck.getCard(ACECLUBS_LOC));
163         assertEquals(new Card(ACE, HEARTS),
164                      defaultDeck.getCard(ACEHEARTS_LOC));
165         // Swap ace of hearts with ace of clubs.
166         Card cardHolder = defaultDeck.getCard(ACECLUBS_LOC);
167         defaultDeck.setCard(ACECLUBS_LOC, defaultDeck.getCard(ACEHEARTS_LOC));
168         defaultDeck.setCard(ACEHEARTS_LOC, cardHolder);
169         int newAceClubsLoc = ACEHEARTS_LOC;
170         int newAceHeartsLoc = ACECLUBS_LOC;
171         assertEquals(Deck.DECK_SIZE, defaultDeck.size());
172         assertEquals(ACE, defaultDeck.getCard(newAceClubsLoc).rank());
173         assertEquals(CLUBS, defaultDeck.getCard(newAceClubsLoc).suit());
174         assertEquals(ACE, defaultDeck.getCard(newAceHeartsLoc).rank());
175         assertEquals(HEARTS, defaultDeck.getCard(newAceHeartsLoc).suit());
176         // Swap four of spades with ace of hearts (at its new location).
177         int fourSpadesLoc = (SPADES.ordinal() * Rank.values().length)
178                             + FOUR.ordinal();
179         cardHolder = defaultDeck.getCard(fourSpadesLoc);
180         defaultDeck.setCard(fourSpadesLoc,
181                             defaultDeck.getCard(newAceHeartsLoc));
182         defaultDeck.setCard(newAceHeartsLoc, cardHolder);
183         assertEquals(FOUR, defaultDeck.getCard(newAceHeartsLoc).rank());
184         assertEquals(SPADES, defaultDeck.getCard(newAceHeartsLoc).suit());
185         assertEquals(ACE, defaultDeck.getCard(fourSpadesLoc).rank());
186         assertEquals(HEARTS, defaultDeck.getCard(fourSpadesLoc).suit());
187     }
