001 package swefun.cardgames.playingcards;
002 /**
003  * Model cards as from a deck of playing cards.
004  * A card is represented by a Rank+Suit pair.
005  * Each unique Rank+Suit pair is also identified by
006  * a unique integer value, referred to as its ID,
007  * which may be used for new card construction.
008  *
009  * @author Dr. Jody Paul
010  * @version 1.6 $Id: Card.java 295 2017-03-24 21:26:19Z jody $
011  */
012 public class Card {
013     /** The rank of this card. */
014     private Rank rank;
015 
016     /** The suit of this card. */
017     private Suit suit;
018 
019     /** An integer uniquely identifying this rank-suit pair. */
020     private int myCardID;
021 
022     /**
023      * Constructor for a card with arbitrary rank and suit.
024      */
025     public Card() {
026         this(Rank.valueOf("QUEEN"), Suit.valueOf("HEARTS"));
027     }
028 
029     /**
030      * Card constructor.
031      * @param cardRank the rank for this card
032      * @param cardSuit the suit for this card
033      */
034     public Card(final Rank cardRank, final Suit cardSuit) {
035         this.rank = cardRank;
036         this.suit = cardSuit;
037         this.myCardID = generateCardID(this.rank, this.suit);
038     }
039 
040     /**
041      * Card copy constructor.
042      * @param card the card to be copied
043      */
044     public Card(final Card card) {
045         this(card.rank, card.suit);
046     }
047 
048     /**
049      * Card constructor from a card ID.
050      * @param cardID the ID of the card to construct
051      */
052     public Card(final int cardID) {
053         this(retrieveRank(cardID), retrieveSuit(cardID));
054     }
055 
056     /**
057      * Utility to generate an integer card ID corresponding
058      * to the specified Rank+Suit pair.
059      * <P><CODE>
060      * cardID = (cardSuit.ordinal() * Rank.values().length))
061      *          + cardRank.ordinal()
062      * </CODE></P>
063      * @param cardRank the rank of the card
064      * @param cardSuit the suit of the card
065      * @return the card ID associated with the given rank and suit
066      */
067     public static int generateCardID(final Rank cardRank, final Suit cardSuit) {
068         int cardID = cardSuit.ordinal() * Rank.values().length;
069         cardID += cardRank.ordinal();
070         return cardID;
071     }
072 
073     /**
074      * Utility to retrieve rank from an integer card ID.
075      * @param cardID the card ID to process
076      * @return the rank associated with the given card ID
077      */
078     public static Rank retrieveRank(final int cardID) {
079         return Rank.values()[cardID % Rank.values().length];
080     }
081 
082     /**
083      * Utility to retrieve suit from an integer card ID.
084      * @param cardID the card ID to process
085      * @return the suit associated with the given card ID
086      */
087     public static Suit retrieveSuit(final int cardID) {
088         return Suit.values()[(int) Math.floor(cardID / Rank.values().length)];
089     }
090 
091     /**
092      * Access the rank of this card.
093      * @return the rank of this card
094      */
095     public Rank rank() {
096         return this.rank;
097     }
098 
099     /**
100      * Access the suit of this card.
101      * @return the suit of this card
102      */
103     public Suit suit() {
104         return this.suit;
105     }
106 
107     /**
108      * Access the ID of this card.
109      * @return the ID of this card
110      */
111     public int cardID() {
112         return this.myCardID;
113     }
114 
115     /**
116      * Render this card as commonly used phrase:
117      *   <i>rank</i> of <i>suit</i>.
118      * Examples: "Five of Diamonds", "King of Hearts"
119      * @return display name of this card
120      */
121     @Override
122     public String toString() {
123         return this.rank + " of " + this.suit;
124     }
125 
126     /**
127      * Overridden equals method;
128      *   both rank and suit must be the same.
129      * @param obj the object to compare with this card
130      * @return true if and only if rank and suit match
131      */
132     @Override
133     public boolean equals(final Object obj) {
134         if (obj == null) {
135             return false;
136         }
137         if (!(obj.getClass().equals(this.getClass()))) {
138             return false;
139         }
140         Card cobj = (Card) obj;
141         return (this.rank.equals(cobj.rank) && this.suit.equals(cobj.suit));
142     }
143 
144     @Override
145     public int hashCode() {
146         return (this.rank.hashCode() + this.suit.hashCode());
147     }
148 
149     /**
150      * Internal testing.
151      * @param args ignored
152      */
153     public static void main(final String[] args) {
154         Card c = new Card();
155         Card d = new Card(c);
156         assert (c != d);
157         assert (c.equals(d));
158         assert (c.cardID() == d.cardID());
159     }
160 }
