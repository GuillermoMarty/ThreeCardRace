003 import java.util.List;
004 import java.util.ArrayList;
005 import java.util.Iterator;
006 /**
007  * A hand comprised of cards.
008  * A hand is also represented by a unique long integer value.
009  * @author Dr. Jody Paul
010  * @version 1.4 $Id: Hand.java 295 2017-03-24 21:26:19Z jody $
011  */
012 public class Hand implements Iterable<Card> {
013     /** The default number of cards in a hand. */
014     static final int HAND_SIZE = 5;
015 
016     /** Cards in this hand. */
017     private final List<Card> handList = new ArrayList<Card>();
018 
019     /**
020      * Construct a hand of default size from a default deck of cards.
021      */
022     public Hand() {
023         this(HAND_SIZE);
024     }
025 
026     /**
027      * Construct a hand of specified size from a default deck of cards.
028      * @param handSize the number of cards in this hand
029      */
030     public Hand(final int handSize) {
031         this(handSize, new Deck());
032     }
033 
034     /**
035      * Construct a hand from the beginning of a given deck.
036      * @param handSize the number of cards in this hand
037      * @param fromDeck the deck from which to draw cards
038      */
039     public Hand(final int handSize, final Deck fromDeck) {
040         this(handSize, fromDeck, 0);
041     }
042 
043     /**
044      * Construct a hand given a deck and a starting position
045      *     within that deck.
046      * @param handSize the number of cards in this hand
047      * @param fromDeck the deck from which to draw cards
048      * @param start the starting index of cards in the deck
049      */
050     public Hand(final int handSize, final Deck fromDeck, final int start) {
051         for (int i = 0; i < handSize; i++) {
052             this.handList.add(fromDeck.getCard(i + start));
053         }
054     }
055 
056     /**
057      * Create a new hand given a hand encoding.
058      * @param encoding the encoding of the desired hand
059      */
060     public Hand(final long encoding) {
061         long carrier = encoding;
062         int  currentCard;
063 
064         for (int i = HAND_SIZE - 1; i >= 0; i--) {
065             currentCard = (int) carrier % Deck.ENCODER;
066             this.handList.add(new Card(currentCard));
067             carrier = (carrier - currentCard) / Deck.ENCODER;
068         }
069     }
070 
071     /**
072      * Hand copy constructor.
073      * Uses references to the original cards.
074      * @param orig the hand to be copied
075      */
076     public Hand(final Hand orig) {
077         for (Card c : orig) {
078             this.handList.add(c);
079         }
080     }
081 
082     @Override
083     public Iterator<Card> iterator() {
084         return handList.iterator();
085     }
086 
087     /**
088      * Two hands are considered equivalent if they both
089      * contain equivalent cards in the same order.
090      * @param obj the object to compare with this hand
091      * @return true if and only if both hands have the same
092      *         cards in the same order
093      */
094     @Override
095     public boolean equals(final Object obj) {
096         if (obj == null) {
097             return false;
098         }
099         if (!(obj.getClass().equals(this.getClass()))) {
100             return false;
101         }
102         Hand hobj = (Hand) obj;
103         for (int i = 0; i < getSize(); i++) {
104             if (!this.handList.get(i).equals(hobj.handList.get(i))) {
105                 return false;
106             }
107         }
108         return true;
109     }
110 
111     @Override
112     public int hashCode() {
113         int code = 0;
114         for (Card c : this.handList) {
115             code += c.hashCode();
116         }
117         return code;
118     }
119 
120     /**
121      * Retrieve the card at a given position within this hand.
122      * <p><b>Precondition: </b>
123      *     <CODE>0 &le; which &le; </CODE>size of hand</p>
124      * @param which the index of the desired card
125      * @return the card at the specified index
126      *         or at index 0 if precondition is not met
127      */
128     public final Card getCard(final int which) {
129         if (0 <= which && which < this.handList.size()) {
130             return this.handList.get(which);
131         }
132         return this.handList.get(0);
133     }
134 
135     /**
136      * Set a specified card of this hand.
137      * <p><b>Precondition: </b>
138      *     <CODE>0 &le; which &le; </CODE>size of hand</p>
139      * @param which the index of the card to change
140      * @param newCard the card to be placed at the specified index
141      */
142     public final void setCard(final int which, final Card newCard) {
143         if (0 <= which && which < this.handList.size()) {
144             handList.set(which, newCard);
145         }
146     }
147 
148     /**
149      * Access the number of cards that make up this hand.
150      * @return the size of this hand
151      */
152     public int getSize() {
153         return this.handList.size();
154     }
155 
156     /**
157      * Sort this hand by ascending card rank; ignores suit.
158      */
159     public void sort() {
160         int min;
161         Card temp;
162 
163         for (int i = 0; i < this.handList.size(); i++) {
164             min = i;
165             for (int j = i; j < this.handList.size(); j++) {
166                 if ((this.handList.get(j))
167                          .rank()
168                          .compareTo((this.handList.get(min)).rank()) < 0) {
169                     min = j;
170                 }
171             }
172             temp = this.handList.get(i);
173             this.handList.set(i, this.handList.get(min));
174             this.handList.set(min, temp);
175         }
176     }
177 
178     /**
179      * Encode this hand as an integer value.
180      * Uses encoding scheme determined by the {@link Deck} class.
181      * @return the encoding of this hand
182      */
183     public final long encode() {
184         long encoding = 0;
185         for (int i = this.handList.size() - 1; i >= 0; i--) {
186             encoding = (Deck.ENCODER * encoding)
187                        + this.handList.get(i).cardID();
188         }
189         return encoding;
190     }
191 
192     /**
193      * Render this hand as a string of delimited card names.
194      * @return displayable string for this hand
195      */
196     @Override
197     public String toString() {
198         String s = "";
199         for (int i = 0; i < this.handList.size(); i++) {
200             s += "[" + handList.get(i) + "]";
201         }
202         return s;
203     }
