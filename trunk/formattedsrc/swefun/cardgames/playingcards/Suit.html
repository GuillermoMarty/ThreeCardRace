Suit.java
01 package swefun.cardgames.playingcards;
02 /**
03  * Enumeration class for the suits of cards.
04  * Supports a color() method that retrieves the
05  * color string associated with a suit:
06  * Clubs = "BLACK", Diamonds = "RED",
07  * Hearts = "RED", Spades = "BLACK".
08  * @author Dr. Jody Paul
09  * @version 1.2 $Id: Suit.java 295 2017-03-24 21:26:19Z jody $
10  */
11 public enum Suit {
12     CLUBS("Clubs", "BLACK"),
13     DIAMONDS("Diamonds", "RED"),
14     HEARTS("Hearts", "RED"),
15     SPADES("Spades", "BLACK");
16 
17     /** The display name of this suit. */
18     private String name;
19 
20     /** The color of this suit. */
21     private String suitColor;
22 
23     /** The color red. */
24     public static final String RED = "RED";
25 
26     /** The color black. */
27     public static final String BLACK = "BLACK";
28 
29     /**
30      * Construct Suit.
31      * @param displayName the display name of this suit
32      * @param color the color of this suit
33      */
34     Suit(final String displayName, final String color) {
35         this.name = displayName;
36         if (color.compareToIgnoreCase(RED) == 0) {
37             this.suitColor = RED;
38         } else if (color.compareToIgnoreCase(BLACK) == 0) {
39             this.suitColor = BLACK;
40         } else {
41             throw (new IllegalArgumentException("Color must be "
42                                                  + BLACK + " or "
43                                                  + RED + "(found "
44                                                  + color + ")"));
45         }
46     }
47 
48     /**
49      * Access the color of this suit.
50      * @return the color of this suit
51      */
52     public String color() {
53         return this.suitColor;
54     }
55 
56     @Override
57     public String toString() {
58         return this.name;
59     }
60 }
