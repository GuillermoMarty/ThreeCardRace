001 package swefun.cardgames.playingcards;
002 
003 import java.util.List;
004 import java.util.ArrayList;
005 import java.util.Iterator;
006 /**
007  * Represents a deck of playing cards.
008  * Position indices are zero-based, such that index 0
009  * represents the "top" of a non-empty deck.
010  * @author Dr. Jody Paul
011  * @version 1.4 $Id: Deck.java 295 2017-03-24 21:26:19Z jody $
012  */
013 public class Deck implements Iterable<Card> {
014     /** Number of cards in a standard deck. */
015     public static final int DECK_SIZE
016               = Suit.values().length * Rank.values().length;
017 
018     /** Minimum number of bits needed to encode a unique card. */
019     private static final int MIN_BITS
020         = (int) Math.ceil(Math.log((double) DECK_SIZE) / Math.log(2.0));
021 
022     /**
023      * Encoding constant used in deck-encoding formula:
024      * &sum; cardID[i] * ENCODER<SUP>i</SUP> ,&nbsp;
025      *    &forall;i &#8717; 0 &le; i &lt; DECK_SIZE
026      * .
027      */
028     static final int ENCODER = (int) Math.pow(2, MIN_BITS);
029 
030     /** List of cards in this deck. */
031     private final List<Card> cardList = new ArrayList<Card>(DECK_SIZE);
032 
033     /** Constructs an ordered deck of cards. */
034     public Deck() {
035         for (Suit s : Suit.values()) {
036             for (Rank r : Rank.values()) {
037                 this.cardList.add(new Card(r, s));
038             }
039         }
040     }
041 
042     /**
043      * Constructs a deck with references to cards
044      * provided by the parameter.
045      * @param cards the cards forming the new deck
046      */
047     public Deck(final List<Card> cards) {
048         for (Card c : cards) {
049                 this.cardList.add(c);
050         }
051     }
052 
053     /**
054      * Constructs a deck of cards with references
055      * to cards provided by the parameter.
056      * @param cards the cards forming the new deck
057      */
058     public Deck(final Card[] cards) {
059         for (Card c : cards) {
060                 this.cardList.add(c);
061         }
062     }
063 
064     /**
065      * Returns the number of cards in this deck.
066      * @return the size of this deck
067      */
068     public int size() {
069         return this.cardList.size();
070     }
071 
072     /**
073      * Returns the card at a specified position
074      * in this deck.
075      * @param which the position of the desired card
076      * @return the card at the specified position
077      */
078     public Card getCard(final int which) {
079         return cardList.get(which);
080     }
081 
082     /**
083      * Adds the specified card to the bottom of this deck.
084      * Note that this makes no requirement as to whether or not
085      * the specified card already exists in the deck;
086      * thus, duplicate references may result.
087      * @param newCard the card to be added
088      * @return true if the contents of the deck was changed
089      */
090     public boolean add(final Card newCard) {
091         return cardList.add(newCard);
092     }
093 
094     /**
095      * Inserts the specified card at the specified position
096      * in this deck.
097      * Shifts the card at the specified index (if any) and
098      * any subsequent cards one position to the right (increments
099      * their indices by one).
100      * @param index index at which the specified card is to be inserted
101      * @param newCard the card to be added
102      * @throws IndexOutOfBoundsException if the index is out of range
103      *             <CODE>(index &lt; 0 || index &gt; this.size())</CODE>
104      */
105     public void add(final int index, final Card newCard) {
106         cardList.add(index, newCard);
107     }
108 
109     /**
110      * Removes the specified card from this deck if it is present.
111      * If this deck does not contain the card, the deck is unchanged.
112      * <P>More formally, removes the card with the lowest
113      * index <CODE>i</CODE> such that
114      * <CODE>(card==null ? get(i)==null : card.equals(get(i)))</CODE>
115      * (if such an element exists).
116      * </P>
117      * <P>Returns true if this list contained the specified element
118      * (or equivalently, if this list changed as a result of the call).
119      * </P>
120      * @param card the card to be removed
121      * @return true if the contents of the deck was changed
122      */
123     public boolean remove(final Card card) {
124         return cardList.remove(card);
125     }
126 
127     /**
128      * Removes the card at the specified position
129      * in this deck.
130      * Shifts any subsequent cards one position to the left
131      * (decrements their indices by one).
132      * Returns the card that was removed from the list.
133      * @param index index of the card to be removed
134      * @return the card previously at the specified position
135      * @throws IndexOutOfBoundsException if the index is out of range
136      *             <CODE>(index &lt; 0 || index &gt;= this.size())</CODE>
137      */
138     public Card remove(final int index) {
139             return cardList.remove(index);
140     }
141 
142     /**
143      * Specifies the card at a given position by exchanging cards.
144      * Puts the card currently at that position into the position
145      * vacated by the specified card.
146      * Behavior maintains the integrity of the deck;
147      * that is, the deck contains the same cards before and
148      * after invocation of this method.
149      * <P><B>Preconditions:</B>
150      * <UL>
151      * <LI>The specified card must already exist within the deck.</LI>
152      * <LI>The specified position must be valid</LI>
153      * </UL>
154      * Behavior when preconditions are not met is undefined.
155      * Exceptions may be thrown if preconditions are not met.
156      *
157      *  @param  which   Location in which to put specified card
158      *  @param  inCard  The specified Card
159      *  @return The card now at the indicated position
160      */
161     public Card setCard(final int which, final Card inCard) {
162         // Find inCard's current position.
163         int inCardIndex = this.cardList.indexOf(inCard);
164         // Swap the two cards.
165         Card tempCard = this.cardList.get(which);
166         this.cardList.set(which, this.cardList.get(inCardIndex));
167         this.cardList.set(inCardIndex, tempCard);
168         return this.cardList.get(which);
169     }
170 
171     @Override
172     public Iterator<Card> iterator() {
173         return this.cardList.iterator();
174     }
175 
176     /**
177      * Removes the card from the beginning of the deck.
178      * @return the card that was at the beginning of the deck;
179      *         null if deck contains no cards
180      */
181     public Card deal() {
182         if (this.cardList.size() < 1) {
183             return null;
184         }
185         return this.cardList.remove(0);
186     }
187     /**
188      * Performs a simple shuffle by exchanging cards.
189      * Uses {@link java.util.Collections#shuffle(List)}
190      */
191     public void shuffle() {
192         java.util.Collections.shuffle(this.cardList);
193     }
194 
195     /**
196      * Returns string representation of every card in this deck
197      * and in the order they appear in the deck.
198      * @return displayable string for this deck
199      */
200     @Override
201     public String toString() {
202         String deckString = "";
203         for (Card c : this.cardList) {
204             deckString += c + "   ";
205         }
206         return deckString;
207     }

209     /**
210      * Overridden equals method:
211      *   number and order of equivalent cards must be the same.
212      * @param obj the object to compare with this deck
213      * @return true if and only all if cards match in order
214      */
215     @Override
216     public boolean equals(final Object obj) {
217         if (obj == null) {
218             return false;
219         }
220         if (!(obj.getClass().equals(this.getClass()))) {
221             return false;
222         }
223         Deck dobj = (Deck) obj;
224         boolean equivalent = true;
225         if (dobj.size() == this.size()) {
226             for (int i = 0; i < this.size(); i++) {
227                 if (!dobj.getCard(i).equals(this.getCard(i))) {
228                     equivalent = false;
229                 }
230             }
231         } else {
232             equivalent = false;
233         }
234         return equivalent;
235     }
236 
237     @Override
238     public int hashCode() {
239         long encoding = 0;
240         for (Card c : this.cardList) {
241             encoding = (ENCODER * encoding) + c.cardID();
242         }
243         return (new Long(encoding).intValue());
244     }
