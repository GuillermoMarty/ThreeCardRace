01 package swefun.cardgames.playingcards;
02 
03 /**
04  * Enumeration class for the ranks of cards.
05  * Supports a value() method that returns
06  * integer values: Ace = 1, ..., King = 13
07  * @author Dr. Jody Paul
08  * @version 1.2 $Id: Rank.java 295 2017-03-24 21:26:19Z jody $
09  */
10 public enum Rank {
11     ACE("Ace", 1),
12     DEUCE("Deuce", 2),
13     TREY("Trey", 3),
14     FOUR("Four", 4),
15     FIVE("Five", 5),
16     SIX("Six", 6),
17     SEVEN("Seven", 7),
18     EIGHT("Eight", 8),
19     NINE("Nine", 9),
20     TEN("Ten", 10),
21     JACK("Jack", 11),
22     QUEEN("Queen", 12),
23     KING("King", 13);
24 
25     /** The display name of this rank. */
26     private String name;
27 
28     /** The integer value of this rank. */
29     private int intRank;
30 
31     /**
32      * Construct Rank.
33      * @param displayName the display name of this rank
34      * @param value the integer value of this rank
35      */
36     Rank(final String displayName, final int value) {
37         this.name = displayName;
38         this.intRank = value;
39     }
40 
41     /**
42      * Access the integer value of this rank.
43      * @return the integer value of this rank
44      */
45     public int value() {
46         return this.intRank;
47     }
48 
49     @Override
50     public String toString() {
51         return this.name;
52     }
53 }
